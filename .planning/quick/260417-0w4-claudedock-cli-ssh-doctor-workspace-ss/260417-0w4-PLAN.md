```go
type FileReport struct {
    Path       string // 完整路径，如 /workspace/.ssh/vision
    Kind       string // "private" | "public" | "authorized_keys" | "known_hosts" | "config" | "other"
    Owner      string // "workspace:workspace" 或 "root:root"

    User    string       // 远端 $USER
```

**（1.1）远端扫描脚本（单次 SSH session 执行，避免多轮 round-trip）**：

通过 `sess.Run` 跑一段 bash 脚本，脚本职责：
```
set -u
D=/workspace/.ssh
echo "USER=$(id -un)"
echo "SUDO_OK=$(sudo -n true 2>/dev/null && echo y || echo n)"
[ -d "$D" ] || { echo "SSHDIR_MISSING=$D"; exit 0; }
cd "$D" || exit 0
# 对每个常规文件输出：FILE:<path>\t<owner>\t<mode>\t<firstline>\t<lastbyte_hex>
for f in $(find . -maxdepth 1 -type f | sort); do
  rel="${f#./}"
  path="$D/$rel"
  owner=$(stat -c '%U:%G' "$path" 2>/dev/null || echo "?:?")
  mode=$(stat -c '%a' "$path" 2>/dev/null || echo "?")
  first=$(head -n 1 "$path" 2>/dev/null | tr -d '\r')
  last=$(tail -c 1 "$path" 2>/dev/null | od -An -tx1 | tr -d ' \n')
  # 用 | 分隔，字段本身不含 |；first 字段可能含空格但不会含 |
  printf 'FILE|%s|%s|%s|%s|%s\n' "$rel" "$owner" "$mode" "$first" "$last"
done
```

Go 侧解析这段 stdout，构造 `[]FileReport`。

**（1.2）Kind 判定优先级**：
1. 文件名完全等于 `authorized_keys` → `"authorized_keys"`
2. 文件名完全等于 `known_hosts` 或 `known_hosts.old` → `"known_hosts"`
3. 文件名完全等于 `config` → `"config"`
4. 文件名以 `.pub` 结尾 → `"public"`
5. firstline 以 `-----BEGIN ` 开头且包含 `PRIVATE KEY-----` → `"private"`
6. 其他 → `"other"`

**（1.3）期望 mode 表**（expectedMode 纯函数）：
- private / known_hosts / authorized_keys / config → `0600`
- public → `0644`
- other → 不判定（返回空串，`ModeOK` 置 true 表示不适用）

**（1.4）PEM 尾换行**（仅 `private`）：
- `lastByteHex == "0a"` → `PEMEndsNL = &true`；否则 `&false`。

**（1.5）OwnerOK**：
- `stat` 返回的 `owner` 的前半段（`:` 前）等于远端 `$USER` → OK。

**（1.6）`--fix` 行为**（当 `opts.Fix == true`）：

对每个不 OK 的 FileReport，逐条尝试修复并把结果写入 `FixApplied` / `FixFailed`。所有修复都通过新的 SSH session 执行单行命令（避免 shell 拼接，path 走 stdin 或 printf %q 转义：建议用 `shellescape.Quote` —— 项目已引入 `al.essio.dev/pkg/shellescape`）：

- Owner 不符：`sudo -n chown $USER:$USER <shellescape(path)>`。失败（非零退出或 stderr 非空）→ FixFailed，提示"需要 sudo 权限"。
- Mode 不符：`chmod <expected> <shellescape(path)>`（自有文件时不需要 sudo）。如果 chmod 报 "Operation not permitted" 且 sudo_ok，自动 fallback `sudo -n chmod`。
- PEM 尾少 `\n`：`printf '\n' >> <shellescape(path)>`（同样文件属主前提下不需要 sudo；否则 fallback sudo tee -a）。

**（1.7）`Print` 输出**：
仿 `EnvCheckResult.Print`：
- 标题框 "Cloud Claude SSH 密钥体检报告"
- section："目标" 打印 User / SSHDir / FixMode / SudoOK
- section："私钥 / 公钥 / 其他"（按 Kind 分组列出 FileReport，每行 `  ✓/✗  path  owner=..  mode=..  [PEM=EOL|!EOL]`）
- section："修复动作"（仅 FixMode；逐条打印 FixApplied 前加 `  ✓`，FixFailed 前加 `  ✗`）
- 结尾给出一句建议：
  - 全部 OK → `✓ 全部检查通过`
  - 有失败修复 → `⚠ 仍有项目需要手工处理，见上方"修复失败"清单`
  - 有未修复问题且未开 --fix → `建议：claudedock ssh doctor --fix 尝试自动修复`

**（1.8）`RunSSHDoctor` 主入口**：
1. `sshConnect(cfg)` 打开连接；defer close。
2. 跑扫描脚本 → 解析 → 填 Result 字段。
3. 若 `opts.Fix == true`：遍历 Result.Files 做修复，把每条结果 append 到对应 FileReport 的 FixApplied/FixFailed 并更新 Owner/Mode/PEMEndsNL 字段（重跑一次 stat / tail 验证）。
4. 返回 Result 与 nil。

- 所有用户可见字符串中文。

```go
func runSSHDoctor(cmd *cobra.Command, args []string) error {
    fix, _ := cmd.Flags().GetBool("fix")

    cfg, err := claudedock.LoadConfig()
    if err != nil {
        return err
    }

    client := claudedock.NewEntryClient(cfg.Gateway)

    fmt.Println("正在连接云主机...")
    authResp, err := client.AuthenticateAndWait(
        cmd.Context(),
        cfg.ShortID,
        cfg.Password,
        func(msg string) { fmt.Printf("\r%s", msg) },
    )
    if err != nil {
        return fmt.Errorf("认证失败: %w", err)
    }

    fmt.Println("\r正在体检远端 /workspace/.ssh ...")

    sshCfg := claudedock.SSHConfig{
        Host:     authResp.SSHHost,
        Port:     authResp.SSHPort,
        User:     authResp.SSHUser,
        Password: authResp.SSHPass,
    }

    result, err := claudedock.RunSSHDoctor(sshCfg, claudedock.SSHDoctorOptions{Fix: fix})
    if err != nil {
        return fmt.Errorf("体检失败: %w", err)
    }

    fmt.Println()
    result.Print()
    return nil
}
```
